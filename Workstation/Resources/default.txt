CREATE TABLE Stock_Names (Symbol VARCHAR(5) PRIMARY KEY NOT NULL, Company_Name VARCHAR(50) NOT NULL);
CREATE TABLE Current_Stocks (Symbol VARCHAR(5) PRIMARY KEY NOT NULL, Table_Name VARCHAR(50) NOT NULL);
CREATE TABLE Exchanges (Symbol VARCHAR(6) PRIMARY KEY NOT NULL, Exchange_Name VARCHAR(50) NOT NULL);
CREATE TABLE Stock_Template (Date DATE PRIMARY KEY NOT NULL, Open FLOAT NOT NULL, High FLOAT NOT NULL, Low FLOAT NOT NULL, Close FLOAT NOT NULL, Volume INTEGER NOT NULL, Adj_Close FLOAT NOT NULL);
CREATE TABLE Companies (Symbol VARCHAR(5) PRIMARY KEY NOT NULL, Company_Name VARCHAR(50) NOT NULL);
